script가 먼저 실행 되어 body태그안에 어떤요소를 검색하고자한다면
	검색이 불가하여 오류가 발생된다.
	document.getElementById('result').innerHTML="<h1>hello!</h1>";
	
	body태그 안의 요소로 document를 생성하여 window에 완전히 실행 되어졌을때
	script를 실행하도록 하기 위해서
	window.onload= function() {
	document.getElementById('result').innerHTML="<h1>hello!</h1>";
	}
