	 글자간격 :letter-spacing: 10px;
	상태를 나타내는 코드 : 스타일에서지정 
	:link,:visited, :hover, :active
	
	pre{
		width: 40%;
		float : left;
		margin : 3px;
	}
	
	float :left 속성을 취소하기 위해서 clear: both 을 설정
	clear :both 는 블럭 요소에서만 가능하다.
	
	div{
	clear :both ;
		border : 1px solid gray;
		margin : 10px;
	}

	a:link{
		color :red;
		text-decoration: none;
	}
	a:visited{
		color: green;
	}
	a:hover{
		color : gray;
		font-size:1.5em;
		text-decoration: underline;
	}
	a:active{
		color: blue;
		background : pink;
	}
